function throttle(fn,waittime,immediate){
  let perTime = 0;     // 之前的时间
  let context;
  let args
  let timer = null
  return function () {
   let nowTime = +new Date()
   context =  this       // 如果不缓存的话，this将指向window
   args = argument
  clearTimeout(timer)
   if(nowTime-perTime > immediate) {
   func.apply(context,args)
   perTime = nowTime
   } else {
      timer = setTimeout(function(){
         timer = null
         fn.apply(context,args)
      },waittime)
   }
  }
}
